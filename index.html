{"震慑","缴械","技穷","混乱","嘲讽","虚弱","断粮"} MAX_ROUNDS = 8 class BattleEngine: def __init__(self, team_a: Team, team_b: Team, seed: int | None = None, log: bool = False): self.team_a = team_a.clone() self.team_b.defeated() def winner(self) -> str: if self.team_a.defeated() and self.team_b.defeated(): return "draw" if self.team_b.defeated(): return "A" if self.team_a.defeated(): return "B" hp_a = sum(h.hp for h in self.team_a.alive()) hp_b = sum(h.hp for h in self.team_b.alive()) if hp_a == hp_b: return "draw" return "A" if hp_a > hp_b else "B" def run(self) -> Dict: self.apply_command_passives() while self.round <= MAX_ROUNDS and not self.battle_over(): self.log(f"第 {self.round} 回合開始") self.round_start_effects() units = sorted(self.team_a.alive() + self.team_b.alive(), key=lambda h: h.initiative, reverse=True) for actor in units: if not actor.alive() or self.battle_over(): continue self.take_turn(actor) self.round_end() self.round += 1 return { self.apply_buff_skill(hero, skill) def round_start_effects(self) -> None: for hero in self.team_a.alive() + self.team_b.alive(): if hero.has_status("火攻"): dmg = int(120 + hero.max_hp * 0.018) choose_skill(self, actor: Hero) -> Skill | None: if actor.has_status("技穷"): return None active_skills = [s for s in actor.skills if s.type == "active"] for skill in active_skills: rate = skill.trigger_rate rate -= actor.status_value("主动率降低") rate = max(0.0, min(1.0, rate)) if self.rng.random() <= rate: return skill return None def select_targets(self, actor: Hero, target_type: str) -> List[Hero]: own = self.own_team(actor) enemy = self.enemy_team(actor) if actor.has_status("混乱") and target_type not in {"self","team","ally_two"}: pool = [h for h in own.alive() + enemy.alive() if h skill.name) self.try_apply_effect(actor, t, skill) def apply_buff_skill(self, actor: Hero, skill: Skill) -> None: buff = skill.buff if not buff: return targets = self.select_targets(actor, {ce['name']}") def normal_attack(self, actor: Hero) -> None: if actor.has_status("缴械"): self.log(f"{actor.name} 被缴械，无法普攻") return targets = self.select_targets(actor, "one") if not targets: return
